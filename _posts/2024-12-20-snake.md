<script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');
    const GRID_SIZE = 20;
    const GAME_SIZE = 500;
    const GRID_COUNT = GAME_SIZE / GRID_SIZE;

    let snake = [
        { x: 5, y: 5 },
        { x: 4, y: 5 },
        { x: 3, y: 5 }
    ];
    let direction = 'right';
    let nextDirection = 'right';
    let food = { x: 15, y: 15 };
    let score = 0;
    let level = 1;
    let gameSpeed = 150;
    let gameLoop;
    let natureProgress = 0;

    // Level configurations
    const LEVELS = [
        { required: 5, description: "Barren Meadow" },
        { required: 10, description: "First Sprouts" },
        { required: 15, description: "Growing Garden" },
        { required: 20, description: "Blooming Field" },
        { required: 25, description: "Nature's Paradise" }
    ];

    function drawSnake() {
        snake.forEach((segment, index) => {
            // Snake head
            if (index === 0) {
                ctx.fillStyle = '#4a90e2';
                ctx.beginPath();
                ctx.arc(
                    segment.x * GRID_SIZE + GRID_SIZE/2,
                    segment.y * GRID_SIZE + GRID_SIZE/2,
                    GRID_SIZE/2,
                    0,
                    2 * Math.PI
                );
                ctx.fill();
                
                // Eyes
                ctx.fillStyle = 'white';
                let eyeX = segment.x * GRID_SIZE + GRID_SIZE/2;
                let eyeY = segment.y * GRID_SIZE + GRID_SIZE/2;
                ctx.beginPath();
                ctx.arc(eyeX + 3, eyeY - 3, 2, 0, 2 * Math.PI);
                ctx.arc(eyeX + 3, eyeY + 3, 2, 0, 2 * Math.PI);
                ctx.fill();
            } else {
                // Snake body
                ctx.fillStyle = '#4a90e2';
                ctx.fillRect(
                    segment.x * GRID_SIZE,
                    segment.y * GRID_SIZE,
                    GRID_SIZE - 2,
                    GRID_SIZE - 2
                );
            }
        });
    }

    function drawFood() {
        // Apple shape
        ctx.fillStyle = '#ff6b6b';
        ctx.beginPath();
        ctx.arc(
            food.x * GRID_SIZE + GRID_SIZE/2,
            food.y * GRID_SIZE + GRID_SIZE/2,
            GRID_SIZE/2 - 2,
            0,
            2 * Math.PI
        );
        ctx.fill();

        // Leaf
        ctx.fillStyle = '#90ce7f';
        ctx.beginPath();
        ctx.ellipse(
            food.x * GRID_SIZE + GRID_SIZE/2 + 2,
            food.y * GRID_SIZE + 2,
            4,
            6,
            Math.PI/4,
            0,
            2 * Math.PI
        );
        ctx.fill();
    }

    function updateNatureProgress() {
        natureProgress = Math.min(100, (score / LEVELS[LEVELS.length-1].required) * 100);
        document.getElementById('progress').style.width = natureProgress + '%';
        
        // Update level
        let newLevel = 1;
        for (let i = 0; i < LEVELS.length; i++) {
            if (score >= LEVELS[i].required) {
                newLevel = i + 2;
            }
        }
        
        if (newLevel !== level) {
            level = newLevel;
            document.getElementById('level').textContent = level;
            gameSpeed = Math.max(70, 150 - (level - 1) * 15);
            clearInterval(gameLoop);
            gameLoop = setInterval(gameUpdate, gameSpeed);
        }
    }

    function drawBackground() {
        // Background is handled by CSS
        ctx.clearRect(0, 0, GAME_SIZE, GAME_SIZE);
        
        // Draw nature elements based on progress
        if (natureProgress > 20) {
            // Draw scattered flowers
            for (let i = 0; i < natureProgress/5; i++) {
                let x = Math.random() * GAME_SIZE;
                let y = Math.random() * GAME_SIZE;
                if (!isOnSnake(x/GRID_SIZE, y/GRID_SIZE)) {
                    drawFlower(x, y);
                }
            }
        }
    }

    function drawFlower(x, y) {
        const colors = ['#ff6b6b', '#ffd93d', '#6c5ce7', '#a8e6cf'];
        ctx.fillStyle = colors[Math.floor(Math.random() * colors.length)];
        ctx.beginPath();
        for (let i = 0; i < 5; i++) {
            const angle = (i * 2 * Math.PI) / 5;
            const petalX = x + Math.cos(angle) * 5;
            const petalY = y + Math.sin(angle) * 5;
            ctx.arc(petalX, petalY, 3, 0, 2 * Math.PI);
        }
        ctx.fill();
        
        // Flower center
        ctx.fillStyle = '#ffd93d';
        ctx.beginPath();
        ctx.arc(x, y, 2, 0, 2 * Math.PI);
        ctx.fill();
    }

    function isOnSnake(x, y) {
        return snake.some(segment => {
            const snakeX = segment.x * GRID_SIZE;
            const snakeY = segment.y * GRID_SIZE;
            return Math.abs(x*GRID_SIZE - snakeX) < GRID_SIZE && 
                   Math.abs(y*GRID_SIZE - snakeY) < GRID_SIZE;
        });
    }

    function gameUpdate() {
        const head = { x: snake[0].x, y: snake[0].y };
        direction = nextDirection;

        switch(direction) {
            case 'up': head.y--; break;
            case 'down': head.y++; break;
            case 'left': head.x--; break;
            case 'right': head.x++; break;
        }

        // Wall collision
        if (head.x < 0) head.x = GRID_COUNT - 1;
        if (head.x >= GRID_COUNT) head.x = 0;
        if (head.y < 0) head.y = GRID_COUNT - 1;
        if (head.y >= GRID_COUNT) head.y = 0;

        // Self collision
        if (snake.some(segment => segment.x === head.x && segment.y === head.y)) {
            gameOver();
            return;
        }

        snake.unshift(head);

        // Food collision
        if (head.x === food.x && head.y === food.y) {
            score++;
            document.getElementById('score').textContent = score;
            updateNatureProgress();
            placeFood();
        } else {
            snake.pop();
        }

        // Draw game
        drawBackground();
        drawSnake();
        drawFood();
    }

    function placeFood() {
        do {
            food.x = Math.floor(Math.random() * GRID_COUNT);
            food.y = Math.floor(Math.random() * GRID_COUNT);
        } while (snake.some(segment => segment.x === food.x && segment.y === food.y));
    }

    function gameOver() {
        clearInterval(gameLoop);
        alert(`Game Over! You restored ${Math.floor(natureProgress)}% of nature!`);
        resetGame();
    }

    function resetGame() {
        snake = [
            { x: 5, y: 5 },
            { x: 4, y: 5 },
            { x: 3, y: 5 }
        ];
        direction = 'right';
        nextDirection = 'right';
        score = 0;
        level = 1;
        natureProgress = 0;
        gameSpeed = 150;
        document.getElementById('score').textContent = score;
        document.getElementById('level').textContent = level;
        document.getElementById('progress').style.width = '0%';
        placeFood();
        gameLoop = setInterval(gameUpdate, gameSpeed);
    }

    document.addEventListener('keydown', (e) => {
        switch(e.key) {
            case 'ArrowUp':
                if (direction !== 'down') nextDirection = 'up';
                break;
            case 'ArrowDown':
                if (direction !== 'up') nextDirection = 'down';
                break;
            case 'ArrowLeft':
                if (direction !== 'right') nextDirection = 'left';
                break;
            case 'ArrowRight':
                if (direction !== 'left') nextDirection = 'right';
                break;
        }
    });

    // Start game
    resetGame();
</script>
